```python
class LotteryDraw(BaseModel):
    round_number: int
    machine: Literal["1호기", "2호기", "3호기"]
    numbers: list[int]  # 6 sorted unique ints, 1-45
    odd_even_ratio: str
    high_low_ratio: str
    ac_value: int
    tail_sum: int
    total_sum: int

class MachineDataResponse(BaseModel):
    machine: str
    total_draws: int
    draws: list[LotteryDraw]

class PredictRequest(BaseModel):
    machine: Literal["1호기", "2호기", "3호기"]
    strategy: Literal["frequency", "pattern", "range", "balance", "composite"]

class PredictResponse(BaseModel):
    games: list[list[int]]  # 5 games, each 6 sorted unique numbers
    strategy: str
    machine: str
```
